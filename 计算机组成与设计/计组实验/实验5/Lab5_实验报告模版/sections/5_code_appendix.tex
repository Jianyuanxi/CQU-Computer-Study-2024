\section{i\_cache 代码}

\begin{lstlisting}[language=Verilog]
module i_cache (
    input wire clk, rst,
    input         cpu_inst_req     ,
    input         cpu_inst_wr      ,
    input  [1 :0] cpu_inst_size    ,
    input  [31:0] cpu_inst_addr    ,
    input  [31:0] cpu_inst_wdata   ,
    output [31:0] cpu_inst_rdata   ,
    output        cpu_inst_addr_ok ,
    output        cpu_inst_data_ok ,
    output         cache_inst_req     ,
    output         cache_inst_wr      ,
    output  [1 :0] cache_inst_size    ,
    output  [31:0] cache_inst_addr    ,
    output  [31:0] cache_inst_wdata   ,
    input   [31:0] cache_inst_rdata   ,
    input          cache_inst_addr_ok ,
    input          cache_inst_data_ok
);
parameter INDEX_WIDTH  = 10;
parameter OFFSET_WIDTH = 2;
localparam TAG_WIDTH   = 32 - INDEX_WIDTH - OFFSET_WIDTH;
localparam DEPTH       = 1 << INDEX_WIDTH;

reg                 valid    [DEPTH-1:0];
reg [TAG_WIDTH-1:0] tag_arr  [DEPTH-1:0];
reg [31:0]          data_arr [DEPTH-1:0];

integer i;
always @(posedge clk) begin
    if (rst)
        for (i = 0; i < DEPTH; i = i+1) valid[i] <= 1'b0;
end

wire [INDEX_WIDTH-1:0] req_index = cpu_inst_addr[INDEX_WIDTH+OFFSET_WIDTH-1:OFFSET_WIDTH];
wire [TAG_WIDTH-1:0]   req_tag   = cpu_inst_addr[31:INDEX_WIDTH+OFFSET_WIDTH];
wire hit = valid[req_index] & (tag_arr[req_index] == req_tag);

localparam IDLE = 1'b0, RM = 1'b1;
reg state;
reg [INDEX_WIDTH-1:0] save_index;
reg [TAG_WIDTH-1:0]   save_tag;
reg [31:0]            save_addr;
reg [1:0]             save_size;

always @(posedge clk) begin
    if (rst) begin
        state <= IDLE;
    end else case (state)
        IDLE: begin
            if (cpu_inst_req & ~hit) begin
                state      <= RM;
                save_index <= req_index;
                save_tag   <= req_tag;
                save_addr  <= cpu_inst_addr;
                save_size  <= cpu_inst_size;
            end
        end
        RM: if (cache_inst_data_ok) state <= IDLE;
    endcase
end

reg addr_rcv;
always @(posedge clk) begin
    if (rst)
        addr_rcv <= 1'b0;
    else if (state == IDLE)
        addr_rcv <= 1'b0;
    else if (cache_inst_req & cache_inst_addr_ok)
        addr_rcv <= 1'b1;
end

always @(posedge clk) begin
    if (!rst && state == RM && cache_inst_data_ok) begin
        valid   [save_index] <= 1'b1;
        tag_arr [save_index] <= save_tag;
        data_arr[save_index] <= cache_inst_rdata;
    end
end

assign cpu_inst_rdata   = (state==RM) ? cache_inst_rdata : data_arr[req_index];
assign cpu_inst_addr_ok = (cpu_inst_req & hit) |
                          (state==RM & cache_inst_addr_ok & ~addr_rcv);
assign cpu_inst_data_ok = (cpu_inst_req & hit) |
                          (state==RM & cache_inst_data_ok);

assign cache_inst_req   = (state == RM) & ~addr_rcv;
assign cache_inst_wr    = 1'b0;
assign cache_inst_size  = (state==RM) ? save_size : cpu_inst_size;
assign cache_inst_addr  = (state==RM) ? save_addr : cpu_inst_addr;
assign cache_inst_wdata = 32'b0;
endmodule
\end{lstlisting}

\section{d\_cache 代码}

\begin{lstlisting}[language=Verilog]
module d_cache (
    input wire clk, rst,
    input         cpu_data_req     ,
    input         cpu_data_wr      ,
    input  [1 :0] cpu_data_size    ,
    input  [31:0] cpu_data_addr    ,
    input  [31:0] cpu_data_wdata   ,
    output [31:0] cpu_data_rdata   ,
    output        cpu_data_addr_ok ,
    output        cpu_data_data_ok ,
    output         cache_data_req     ,
    output         cache_data_wr      ,
    output  [1 :0] cache_data_size    ,
    output  [31:0] cache_data_addr    ,
    output  [31:0] cache_data_wdata   ,
    input   [31:0] cache_data_rdata   ,
    input          cache_data_addr_ok ,
    input          cache_data_data_ok
);
parameter INDEX_WIDTH  = 10;
parameter OFFSET_WIDTH = 2;
localparam TAG_WIDTH   = 32 - INDEX_WIDTH - OFFSET_WIDTH;
localparam DEPTH       = 1 << INDEX_WIDTH;

reg                 valid    [DEPTH-1:0];
reg [TAG_WIDTH-1:0] tag_arr  [DEPTH-1:0];
reg [31:0]          data_arr [DEPTH-1:0];

integer i;
always @(posedge clk) begin
    if (rst)
        for (i = 0; i < DEPTH; i = i+1) valid[i] <= 1'b0;
end

wire [INDEX_WIDTH-1:0] req_index = cpu_data_addr[INDEX_WIDTH+OFFSET_WIDTH-1:OFFSET_WIDTH];
wire [TAG_WIDTH-1:0]   req_tag   = cpu_data_addr[31:INDEX_WIDTH+OFFSET_WIDTH];
wire hit      = valid[req_index] & (tag_arr[req_index] == req_tag);
wire is_read  = cpu_data_req & ~cpu_data_wr;
wire is_write = cpu_data_req &  cpu_data_wr;

localparam IDLE=2'b00, RM=2'b01, WM=2'b10;
reg [1:0] state;
reg [INDEX_WIDTH-1:0] save_index;
reg [TAG_WIDTH-1:0]   save_tag;
reg [31:0]            save_addr;
reg [1:0]             save_size;
reg [31:0]            save_wdata;

always @(posedge clk) begin
    if (rst) begin
        state <= IDLE;
    end else case (state)
        IDLE: begin
            if (is_read & ~hit) begin
                state <= RM;
                save_index <= req_index; save_tag   <= req_tag;
                save_addr  <= cpu_data_addr; save_size  <= cpu_data_size;
            end else if (is_write) begin
                state <= WM;
                save_index <= req_index; save_tag   <= req_tag;
                save_addr  <= cpu_data_addr; save_size  <= cpu_data_size;
                save_wdata <= cpu_data_wdata;
            end
        end
        RM: if (cache_data_data_ok) state <= IDLE;
        WM: if (cache_data_data_ok) state <= IDLE;
        default: state <= IDLE;
    endcase
end

reg addr_rcv;
always @(posedge clk) begin
    if (rst)                                      addr_rcv <= 1'b0;
    else if (state == IDLE)                        addr_rcv <= 1'b0;
    else if (cache_data_req & cache_data_addr_ok) addr_rcv <= 1'b1;
end

wire [3:0] wmask =
    (save_size==2'b00) ?
        (save_addr[1] ? (save_addr[0] ? 4'b1000 : 4'b0100)
                      : (save_addr[0] ? 4'b0010 : 4'b0001)) :
    (save_size==2'b01) ? (save_addr[1] ? 4'b1100 : 4'b0011) : 4'b1111;

always @(posedge clk) begin
    if (!rst) begin
        if (state==RM && cache_data_data_ok) begin
            valid   [save_index] <= 1'b1;
            tag_arr [save_index] <= save_tag;
            data_arr[save_index] <= cache_data_rdata;
        end else if (state==WM && cache_data_data_ok &&
                     valid[save_index] && tag_arr[save_index]==save_tag) begin
            data_arr[save_index] <=
                (data_arr[save_index] &
                 ~{{8{wmask[3]}},{8{wmask[2]}},{8{wmask[1]}},{8{wmask[0]}}}) |
                (save_wdata &
                  {{8{wmask[3]}},{8{wmask[2]}},{8{wmask[1]}},{8{wmask[0]}}});
        end
    end
end

assign cpu_data_rdata   = (state==RM) ? cache_data_rdata : data_arr[req_index];
assign cpu_data_addr_ok = (is_read & hit) |
                          ((state==RM||state==WM) & cache_data_addr_ok & ~addr_rcv);
assign cpu_data_data_ok = (is_read & hit) |
                          ((state==RM||state==WM) & cache_data_data_ok);

assign cache_data_req   = (state==RM || state==WM) & ~addr_rcv;
assign cache_data_wr    = (state==WM);
assign cache_data_size  = save_size;
assign cache_data_addr  = save_addr;
assign cache_data_wdata = save_wdata;
endmodule
\end{lstlisting}
