① 返回标量值的函数
语法模板
create function 函数名 (参数名 类型) returns integer -- 返回类型（integer / varchar(n) 等） begin -- 多条语句时必须有 begin-end declare 变量名 类型; select 列名 into 变量名 from 表名 where 条件; return 变量名; end;
⚠️ 函数体只有 一条 SQL 语句时，不需要 begin-end；多条语句必须加。
完整示例 — 返回教师人数
create function instructor_count (min_sal numeric(8,2)) returns integer begin declare cnt integer; select count(*) into cnt from instructor where salary > min_sal; return cnt; end;

调用方式：直接在 SQL 中当函数用 → select instructor_count(50000);

② 返回表的函数
语法模板
create function 函数名 (参数名 类型) returns table (列1 类型, 列2 类型, ...) -- 声明返回表的每一列 return table ( -- 注意：return table，不是 returns select 列1, 列2, ... from 表名 where 条件 );
⚠️ 声明用 returns table(...)，函数体用 return table(...)，两处拼写不同！
完整示例 — 返回高薪教师列表
create function instructors_above_avg () returns table (ID varchar(5), name varchar(20), salary numeric(8,2)) return table ( select ID, name, salary from instructor where salary > (select avg(salary) from instructor) );
调用方式：select * from table(instructors_above_avg());
❌ 错误写法：select * from instructors_above_avg();（少了 table()包裹）
过程语法模板
create procedure 过程名 ( in 参数名 类型, -- 输入参数 out 参数名 类型) -- 输出参数（不需要在 begin-end 里 declare） begin select 列名 into out参数名 from 表名 where 条件; end;
完整示例
定义过程
create procedure get_budget ( in dname varchar(20), out dept_budget numeric(12,2)) begin select budget into dept_budget from department where dept_name = dname; end;
调用过程（调用方需自己 declare）
declare dept_budget numeric(12,2); -- 调用方声明变量接收 out 值 call get_budget('Physics', dept_budget);
过程易错点
❌ out 参数在 begin-end 里重复 declare

create procedure p(out x integer)
begin
declare x integer; -- 错！
...
✅ out 参数直接在 begin-end 里用，无需 declare
❌ 调用方忘记 declare 接收变量

call get_budget('CS', dept_budget); -- 报错，变量未声明
✅ 必须先 declare，再 call

declare dept_budget numeric(12,2);
call get_budget('CS', dept_budget);

触发器语法模板
create trigger 触发器名 after/before insert/delete/update of 表名 on (列名) on 表名 referencing new row as nrow -- insert / update 可用 referencing old row as orow -- delete / update 可用 for each row -- 行级触发（每行触发一次） when (触发条件) begin atomic -- 触发动作 end;
⚠️ update 指定列的写法固定：update of 表名 on (列名)，of 在前 on 在后，不能颠倒！
new row / old row 使用规则
操作类型new row（操作后）old row（操作前）
INSERT可用 ✓不可用 ✗
DELETE不可用 ✗可用 ✓
UPDATE可用 ✓可用 ✓
课件三大案例
案例1：insert 触发 — 插入 section 时检查 time_slot_id 合法性
create trigger timeslot_check1 after insert on section referencing new row as nrow for each row when (nrow.time_slot_id not in ( select time_slot_id from time_slot)) begin rollback; end;
案例2：delete 触发 — 删除 time_slot 时检查 section 里是否还有引用
create trigger timeslot_check2 after delete on time_slot referencing old row as orow for each row when (orow.time_slot_id not in ( select time_slot_id from time_slot) and orow.time_slot_id in ( select time_slot_id from section)) begin rollback; end;
案例3：update 触发 — 成绩从不及格变及格时更新学生总学分（同时用 nrow + orow）
create trigger credits_earned after update of takes on (grade) referencing new row as nrow referencing old row as orow for each row when (nrow.grade <> 'F' and nrow.grade is not null and (orow.grade = 'F' or orow.grade is null)) begin atomic update student set tot_cred = tot_cred + ( select credits from course where course.course_id = nrow.course_id) where student.id = nrow.id; end;

<> 等价于 !=，表示"不等于"，课件用 <> 写法。

触发器易错点汇总
❌ update 触发指定列时，of / on 顺序写反
after update on instructor of (salary) — 错！
✅ 正确：after update of instructor on (salary)
❌ 改粒度时只改了 for each，忘了改 when 里的别名
把 for each statement 改成 for each row 后，when 里也要从 table别名.列 改成 row别名.列
❌ 触发动作里再去 from 表 查询当前行的数据
insert into log select col from 原表 where ... — 逻辑错误！
✅ 当前行已在 nrow/orow 里，直接用 values 插入：
insert into log values (nrow.col1, orow.col2 - nrow.col2);
❌ insert 触发器里声明了 old row — 不合法，insert 前没有旧行
⚠️ begin atomic 与 begin：多条语句用 begin atomic（作为一个事务）；单条可省略 begin-end

语法一句话记忆卡
标量函数
create function → returns 类型 → begin → declare → select into → return → end;
返回表函数
create function → returns table(列声明) → return table(select 查询);
过程
create procedure → (in x, out y) → begin → select into y → end;
调用方：declare y; call p(v, y);
触发器 insert
after insert on 表 → referencing new row as nrow → for each row → when(...) → begin → 动作 → end;
触发器 delete
after delete on 表 → referencing old row as orow → for each row → when(...) → begin → 动作 → end;
触发器 update
after update of 表 on (列) → referencing new row / old row → for each row → when(...) → begin atomic → 动作 → end;
